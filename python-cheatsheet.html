📦 Variables & Data Types
Data Types
# No type declaration needed — Python figures it out name = "Kwame" # str — text age = 21 # int — whole number temp = 36.5 # float — decimal active = True # bool — True or False nothing = None # NoneType — no value # Check type: print(type(age)) # <class 'int'>
Output & f-strings
name = "Ama"; age = 20 print("Hello") print("Name:", name) print(f"Hi {name}, you are {age}") print(f"{age + 1} next year") print(f"{3.14159:.2f}") # 3.14 (2 decimals) print(f"{'hello':>10}") # right-align 10 chars print(f"{'hi':<10}") # left-align 10 chars
User Input
name = input("Your name: ") # always a string age = int(input("Age: ")) # convert to int temp = float(input("Temp: ")) # convert to float # Safe input with error handling: try: n = int(input("Number: ")) except ValueError: print("That's not a number!")
Arithmetic Operators
10 + 3 # 13 addition 10 - 3 # 7 subtraction 10 * 3 # 30 multiplication 10 / 3 # 3.333 division (always float) 10 // 3 # 3 floor division (no decimal) 10 % 3 # 1 modulus (remainder) 10 ** 3 # 1000 power (10 cubed) x += 5 # x = x + 5 (shorthand) x -= 2 # x = x - 2 x *= 3 # x = x * 3
📝 Strings
String Methods
s = " Hello, World! " s.upper() # " HELLO, WORLD! " s.lower() # " hello, world! " s.strip() # "Hello, World!" s.strip().lower() # "hello, world!" s.replace("World", "TechCamp") s.split(",") # [" Hello", " World! "] s.startswith(" Hello") # True s.endswith("!") # True (after strip) s.count("l") # 3 s.find("World") # 9 (index)
Slicing & Indexing
s = "TechCamp" s[0] # "T" (first char) s[-1] # "p" (last char) s[0:4] # "Tech" (chars 0,1,2,3) s[4:] # "Camp" (from 4 to end) s[:4] # "Tech" (from start to 3) s[::-1] # "pmaCcheT" (reversed) len(s) # 8 (length) "Tech" in s # True (substring check) "-".join(["a","b","c"]) # "a-b-c"
Multi-line Strings
text = """This is a multi-line string.""" # Raw string (no escape processing): path = r"C:\Users\Name\file.txt" # Escape characters: # \n = newline # \t = tab # \\ = backslash # \" = quote inside string
Number Formatting
pi = 3.14159265 print(f"{pi:.2f}") # 3.14 print(f"{pi:.4f}") # 3.1416 print(f"{1234567:,}") # 1,234,567 print(f"{0.75:.0%}") # 75% print(f"{42:05d}") # 00042 (zero-padded) print(f"{255:b}") # 11111111 (binary) print(f"{255:x}") # ff (hexadecimal)
📋 Lists, Tuples, Dicts & Sets
Lists
nums = [3, 1, 4, 1, 5] nums.append(9) # add to end nums.insert(0, 0) # insert at index 0 nums.remove(1) # remove first 1 nums.pop() # remove & return last nums.pop(0) # remove & return index 0 nums.sort() # sort ascending nums.sort(reverse=True) # sort descending nums.reverse() # reverse in place nums.index(5) # find index of 5 nums.count(1) # count occurrences of 1 len(nums) # number of items sum(nums) # sum of all numbers min(nums); max(nums) # min and max value
Dictionaries
student = {"name":"Kofi", "age":20} student["name"] # "Kofi" student.get("city", "N/A") # safe get student["city"] = "Accra" # add key student.update({"age":21}) # update del student["city"] # delete key student.keys() # all keys student.values() # all values student.items() # key-value pairs "name" in student # True (key check) student.pop("age") # remove & return
Tuples (immutable lists)
point = (10, 20) # cannot change after creation x, y = point # unpack into variables point[0] # 10 (access by index) len(point) # 2 # Use tuples for: coordinates, RGB colours, # function returning multiple values def minmax(lst): return min(lst), max(lst) # returns tuple lo, hi = minmax([3,1,4,1,5])
Sets (unique values)
a = {1, 2, 3, 3} # {1, 2, 3} — duplicates removed b = {2, 3, 4} a | b # {1,2,3,4} — union a & b # {2,3} — intersection a - b # {1} — difference a.add(5) a.remove(1) len(a) 3 in a # True # Turn list to set to remove duplicates: unique = list(set([1,2,2,3]))
List Comprehensions
# Short way to create lists squares = [x**2 for x in range(1,6)] # [1, 4, 9, 16, 25] evens = [x for x in range(20) if x % 2 == 0] # [0, 2, 4, 6, ... 18] words = ["hi", "hello", "hey"] upper = [w.upper() for w in words] # ["HI", "HELLO", "HEY"]
zip, enumerate, range
# range() range(5) # 0,1,2,3,4 range(1,6) # 1,2,3,4,5 range(0,10,2) # 0,2,4,6,8 (step 2) # enumerate() — get index AND value for i, val in enumerate(["a","b"], 1): print(i, val) # 1 a, 2 b # zip() — loop two lists together names = ["Ama","Kofi"]; scores = [85,92] for n, s in zip(names, scores): print(f"{n}: {s}")
🔀 Control Flow
If / Elif / Else
score = 75 if score >= 80: print("A") elif score >= 70: print("B") # ← this one runs else: print("C or below") # One-liner (ternary): grade = "Pass" if score >= 50 else "Fail" # Comparison operators: # == != > < >= <= # and or not in not in is is not
For Loops
# Loop n times for i in range(5): # 0 to 4 print(i) # Loop through list for name in ["Ama", "Kofi"]: print(name) # Loop through dict d = {"a":1, "b":2} for key, val in d.items(): print(key, val) # Loop through string chars for ch in "Python": print(ch)
While + break / continue
while True: # loop forever x = input("Enter: ") if x == "quit": break # exit loop immediately if x == "": continue # skip this iteration print(f"You typed: {x}") # Loop with else (runs if loop didn't break): for i in range(5): if i == 10: break else: print("Loop finished without break")
Match / Case (Python 3.10+)
day = "Monday" match day: case "Monday" | "Tuesday": print("Early week") case "Friday": print("TGIF!") case _: # default (like else) print("Mid-week") # Like switch/case in other languages
🧩 Functions
Defining Functions
# Basic function def greet(name): print(f"Hello {name}!") # Function with return value def add(a, b): return a + b # Default parameter def power(base, exp=2): return base ** exp power(5) # 25 (uses default exp=2) power(2, 10) # 1024 # Return multiple values (as tuple) def stats(lst): return min(lst), max(lst), sum(lst) lo, hi, total = stats([1,2,3,4,5])
*args & **kwargs
# *args — accept any number of positional args def total(*nums): return sum(nums) total(1, 2, 3, 4) # 10 # **kwargs — accept any keyword arguments def info(**details): for k, v in details.items(): print(f"{k}: {v}") info(name="Ama", age=20, city="Kumasi") # Lambda — one-line anonymous function square = lambda x: x**2 print(square(5)) # 25
Scope
x = 10 # global variable def example(): y = 20 # local — only exists here print(x) # can READ global x global x # declare to WRITE global x x = 99 # now changes global x # Rule: LEGB — Local, Enclosing, Global, Built-in # Python looks up names in this order
Docstrings
def greet(name, greeting="Hello"): """ Greet a person by name. Args: name (str): The person's name. greeting (str): The greeting word. Returns: str: The greeting message. """ return f"{greeting}, {name}!" # Access it: print(greet.__doc__) help(greet)
📁 File Handling & Error Handling
Reading & Writing Files
# Write (creates or overwrites) with open("data.txt", "w") as f: f.write("Line 1\n") f.writelines(["Line 2\n", "Line 3\n"]) # Read whole file with open("data.txt", "r") as f: content = f.read() # whole string lines = f.readlines() # list of lines # Append without overwriting with open("data.txt", "a") as f: f.write("Line 4\n") # Loop through lines with open("data.txt") as f: for line in f: print(line.strip())
Try / Except / Finally
try: n = int(input("Number: ")) result = 100 / n except ValueError: print("Not a number!") except ZeroDivisionError: print("Can't divide by zero!") except Exception as e: print(f"Unexpected: {e}") else: print(f"Result: {result}") # runs if NO error finally: print("Always runs") # runs regardless
os Module — File System
import os os.getcwd() # current directory os.listdir(".") # list files here os.path.exists("x.txt") # True/False os.path.join("dir","file") # safe path join os.makedirs("mydir", exist_ok=True) # create dir os.remove("file.txt") # delete file os.rename("old", "new") # rename os.path.getsize("f.txt") # size in bytes
Raising Exceptions
# Raise your own errors def set_age(age): if age < 0: raise ValueError("Age cannot be negative") if age > 150: raise ValueError("Age too high") return age # assert — for debugging/testing x = 5 assert x > 0, "x must be positive" # AssertionError if x <= 0
